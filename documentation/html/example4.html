A simple example

Wrap both your image and your html map in an angular-maphilight directive: 

<angular-maphilight id="MapTitle" angular-maphilight-config="">
<img class="img-responsive" src="documentation/images/shapes.png" width="500" height="300"/>
<map>
  <area ng-repeat="area in areas" shape="{{area.shape}}" coords="{{area.coords}}"
        title="{{area.title}}" alt="{{area.alt}}"
        href="{{area.link}}" />
</map>
</angular-maphilight>
